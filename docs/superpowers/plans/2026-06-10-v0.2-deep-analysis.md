```python
def check_layer_violations(modules: list[Module], edges: list[Dependency]) -> list[dict]:
    violations = []
    for edge in edges:
        from_mod = _find_module(modules, edge.from_module)
        to_mod = _find_module(modules, edge.to_module)
        if not from_mod or not to_mod:
```

DFS 遍历依赖图检测环：

```python
def detect_circular_dependencies(nodes: list[str], edges: list[Dependency]) -> list[list[str]]:
    """Return list of circular dependency chains."""
    graph: dict[str, set[str]] = {n: set() for n in nodes}
    for edge in edges:
        if edge.from_module in graph and edge.to_module in graph:
            graph[edge.from_module].add(edge.to_module)
    
    cycles = []
    visited = set()
    rec_stack = set()
    path = []
    
    def dfs(node):
        visited.add(node)
        rec_stack.add(node)
        path.append(node)
        for neighbor in graph.get(node, set()):
            if neighbor not in visited:
                dfs(neighbor)
            elif neighbor in rec_stack:
                cycle_start = path.index(neighbor)
                cycles.append(path[cycle_start:])
        path.pop()
        rec_stack.discard(node)
    
    for node in nodes:
        if node not in visited:
            dfs(node)
    return cycles
```

```python
def _infer_business_domains(self) -> list[dict]:
    """Infer business domains from package names and entity names."""
    domains = {}
    keywords = {
        "备份管理": ["backup", "snapshot", "recovery", "restore"],
        "客户端管理": ["client", "agent", "endpoint", "host", "machine"],
        "用户管理": ["user", "account", "auth", "login", "profile"],
        "任务调度": ["task", "schedule", "job", "cron", "trigger"],
        "通知告警": ["notification", "alert", "alarm", "message", "email"],
        "数据统计": ["stat", "report", "dashboard", "usage", "counter"],
        "配置管理": ["config", "setting", "property", "preference"],
        "日志审计": ["log", "audit", "trace", "history"],
    }
    
    for f in self.repo_path.rglob("*.java"):
        try:
            pkg = re.search(r'package\s+([\w.]+);', f.read_text())
            if pkg:
                pkg_name = pkg.group(1).lower()
                for domain, terms in keywords.items():
                    if any(t in pkg_name for t in terms):
                        if domain not in domains:
                            domains[domain] = {"packages": [], "entities": []}
                        domains[domain]["packages"].append(pkg.group(1))
        except: pass
    
    return [{"name": k, **v} for k, v in domains.items()]
```

```python
def _trace_call_chains(self, modules, edges) -> list[dict]:
    """Trace call chains from controllers to repositories."""
    chains = []
    
    # Build adjacency: class_name -> [called_class_names]
    call_graph: dict[str, list[str]] = {}
    class_to_module: dict[str, str] = {}
    
    for f in self.repo_path.rglob("*.java"):
        try:
            content = f.read_text()
            class_name = f.stem
            # Find which module this class belongs to
            for m in modules:
                if any(f.name in mf.get("file","") for mf in m.metadata.get("files",[])):
                    class_to_module[class_name] = m.name
            
            # Find method calls within the class
            calls = set()
            for m in re.finditer(r'(\w+)\.(\w+)\(', content):
                obj = m.group(1)
                if obj in class_to_module:  # Only internal calls
                    calls.add(obj)
            call_graph[class_name] = list(calls)
        except: pass
    
    # Find chains from controllers to repos
    controllers = [m for m in modules if "API_LAYER" in m.metadata.get("role","")]
    for ctrl_mod in controllers:
        for ctrl_file in ctrl_mod.metadata.get("files", []):
            chain = _dfs_call_chain(call_graph, ctrl_file["name"].replace(".java",""), depth=3)
            if chain:
                chains.append(chain)
    
    return chains[:10]  # Top 10 chains
```

```python
def scan_dockerfile(path: Path) -> dict:
    """Parse Dockerfile directives."""
    info = {"base_image": "", "exposed_ports": [], "entrypoint": "", "env_vars": []}
    if not path.exists(): return info
    for line in path.read_text().splitlines():
        line = line.strip()
        if line.startswith("FROM "): info["base_image"] = line[5:]
        elif line.startswith("EXPOSE "): info["exposed_ports"].append(line[7:])
        elif line.startswith("CMD ") or line.startswith("ENTRYPOINT "): info["entrypoint"] += line + " "
        elif line.startswith("ENV "): info["env_vars"].append(line[4:].split()[0])
    return info

def scan_docker_compose(path: Path) -> dict:
    """Parse docker-compose.yml services."""
    import yaml
    try:
        data = yaml.safe_load(path.read_text())
        services = list(data.get("services", {}).keys())
        volumes = list(data.get("volumes", {}).keys())
        return {"services": services, "volumes": volumes}
    except: return {}
```

```python
def compute_complexity(content: str) -> int:
    """Approximate cyclomatic complexity: count branch points."""
    branches = len(re.findall(r'\b(if|for|while|case|catch|\?\s*:)', content))
    return branches + 1

def detect_duplicates(files: list[Path], n_lines: int = 6) -> list[dict]:
    """Detect duplicate code blocks across files."""
    # Simple: hash every N-line window, find collisions
    hashes: dict[str, list[tuple[Path, int]]] = {}
    for f in files:
        try:
            lines = f.read_text().splitlines()
            for i in range(len(lines) - n_lines + 1):
                block = "\n".join(lines[i:i+n_lines]).strip()
                if len(block) > 50:  # Skip tiny blocks
                    h = str(hash(block))
                    if h not in hashes: hashes[h] = []
                    hashes[h].append((f, i+1))
        except: pass
    return [{"files": v, "lines": n_lines} for v in hashes.values() if len(v) > 1][:20]

def comment_ratio(content: str) -> float:
    """Comment lines / total lines."""
    lines = content.splitlines()
    comment_lines = sum(1 for l in lines if l.strip().startswith("//") or l.strip().startswith("/*") or l.strip().startswith("*"))
    return comment_lines / len(lines) if lines else 0
```

```python
def find_long_methods(repo_path: Path, threshold: int = 50) -> list[dict]:
    """Find methods exceeding line count threshold."""
    methods = []
    for f in repo_path.rglob("*.java"):
        try:
            content = f.read_text()
            for m in re.finditer(
                r'(?:public|private|protected)\s+\S+\s+(\w+)\s*\(', content
            ):
                start = m.start()
                # Find matching brace for method body
                body_start = content.find("{", m.end())
                if body_start == -1: continue
                depth = 0
                end = body_start
                for i, ch in enumerate(content[body_start:], body_start):
                    if ch == "{": depth += 1
                    elif ch == "}":
                        depth -= 1
                        if depth == 0:
                            end = i
                            break
                method_lines = content[body_start:end].count("\n")
                if method_lines > threshold:
                    methods.append({
                        "method": m.group(1),
                        "file": str(f.relative_to(repo_path)),
                        "line": content[:start].count("\n") + 1,
                        "loc": method_lines,
                    })
        except: pass
    return sorted(methods, key=lambda m: m["loc"], reverse=True)[:20]
```

```python
def _trace_request_flow(self, controller_class: str, method_name: str) -> list[str]:
    """Trace a request from Controller → Service → Repository."""
    steps = [f"1. {controller_class}.{method_name}()"]
    
    # Find the controller file
    for f in self.repo_path.rglob(f"{controller_class}.java"):
        try:
            content = f.read_text()
            # Find method body
            method_start = content.find(f" {method_name}(")
            if method_start == -1: continue
            
            # Extract service calls
            calls = re.findall(r'(\w+)\.(\w+)\(', content[method_start:])
            for i, (obj, m) in enumerate(calls[:5], 2):
                steps.append(f"{i}. {obj}.{m}()")
        except: pass
    
    return steps
```

```python
    return {
        "total_src": len(src_files),
        "total_tests": len(test_files),
        "covered": covered,
        "uncovered": uncovered,
```
